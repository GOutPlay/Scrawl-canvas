    User interaction - delta animation controls handler

    let updateAnimation = (e) => {

    e.preventDefault();
    e.returnValue = false;

    let val = e.target.value;

    switch (val) {

        case 'off' :
            myLoom.set({ delta: {
                fromPathStart: 0,
                fromPathEnd: 0,
                toPathStart: 0,
                toPathEnd: 0,
            }});
            break;

        case 'posDelta' :
            myLoom.set({ delta: {
                fromPathStart: 0.002,
                fromPathEnd: 0.002,
                toPathStart: 0.002,
                toPathEnd: 0.002,
            }});
            break;

        case 'negDelta' :
            myLoom.set({ delta: {
                fromPathStart: -0.002,
                fromPathEnd: -0.002,
                toPathStart: -0.002,
                toPathEnd: -0.002,
            }});
            break;
    }
};
scrawl.addNativeListener(['input', 'change'], updateAnimation, '#animation');
